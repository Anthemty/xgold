# API Reference

Access GOX's market data, trading functionality, and real-time updates through our comprehensive API.

## Base URLs

```plaintext
Production:  https://api.gox.exchange/v1
Testnet:     https://testnet-api.gox.exchange/v1
WebSocket:   wss://ws.gox.exchange
```

## Authentication

Most endpoints are public and require no authentication. For trading endpoints (coming soon), you'll need an API key.

### Getting an API Key (Coming Soon)

1. Connect wallet on GOX
2. Navigate to **Settings** → **API Keys**
3. Click **"Generate New Key"**
4. Save your API key and secret securely

### Authentication Header

```http
Authorization: Bearer YOUR_API_KEY
```

## Rate Limits

```plaintext
Public Endpoints:  100 requests/minute
Authenticated:     1000 requests/minute
WebSocket:         Unlimited (10 connections max)
```

## Market Data Endpoints

Retrieve current gold spot price (XAU/USD).



### Get Historical Prices

Retrieve historical price data.

```http
GET /market/history/:symbol
```

**Parameters:**
- `symbol` (required): Asset symbol (GOLD, PAXG, XAUT)
- `timeframe` (optional): 1H, 1D, 1W, 1M (default: 1D)
- `from` (optional): Start timestamp (ISO 8601)
- `to` (optional): End timestamp (ISO 8601)
- `limit` (optional): Number of records (max: 2000)

**Example:**

```http
GET /market/history/GOLD?timeframe=1M&from=1978-01-01&limit=500
```

**Response:**

```json
{
  "symbol": "GOLD",
```

## WebSocket API

### Connection

Connect to real-time data feed:

```javascript
const ws = new WebSocket('wss://ws.gox.exchange');

ws.onopen = () => {
  console.log('Connected to GOX WebSocket');
};
```

### Subscribe to Channels

```javascript
// Subscribe to gold price updates
ws.send(JSON.stringify({
  action: 'subscribe',
  channel: 'gold_price'
}));

// Subscribe to token updates
ws.send(JSON.stringify({
  action: 'subscribe',
  channel: 'token',
  symbol: 'PAXG'
}));

// Subscribe to trading pair
ws.send(JSON.stringify({
  action: 'subscribe',
  channel: 'pair',
  pair_id: 'gxau_usdt'
}));
```

### Message Format

**Gold Price Update:**

```json
{
  "channel": "gold_price",
  "type": "price_update",
  "data": {
    "price_usd": 2645.50,
    "change_24h": 12.30,
    "change_pct_24h": 0.47,
    "timestamp": "2025-12-11T10:30:15Z"
  }
}
```

**Token Update:**

```json

  "data": {
```

## Code Examples

### JavaScript/TypeScript

```typescript
// Install axios
// npm install axios

import axios from 'axios';

const BASE_URL = 'https://api.gox.exchange/v1';

// Get gold price
async function getGoldPrice() {
  const response = await axios.get(`${BASE_URL}/market/gold/spot`);
  return response.data;
}

// Get token price
async function getTokenPrice(symbol: string) {
  const response = await axios.get(`${BASE_URL}/market/tokens/${symbol}`);
  return response.data;
}

// Get historical data
async function getHistoricalPrices(symbol: string, timeframe: string = '1D') {
  const response = await axios.get(`${BASE_URL}/market/history/${symbol}`, {
    params: { timeframe, limit: 100 }
  });
  return response.data;
}

// Usage
const gold = await getGoldPrice();
console.log(`Gold: $${gold.price_usd}`);

const paxg = await getTokenPrice('PAXG');
console.log(`PAXG: $${paxg.price_usd} (${paxg.change_pct_24h}%)`);
```

### Python

```python
import requests
import websocket
import json

BASE_URL = 'https://api.gox.exchange/v1'

# Get gold price
def get_gold_price():
    response = requests.get(f'{BASE_URL}/market/gold/spot')
    return response.json()

# Get token price
def get_token_price(symbol):
    response = requests.get(f'{BASE_URL}/market/tokens/{symbol}')
    return response.json()

# WebSocket connection
def on_message(ws, message):
    data = json.loads(message)
    print(f"Received: {data}")

def on_open(ws):
    # Subscribe to gold price
    ws.send(json.dumps({
        'action': 'subscribe',
        'channel': 'gold_price'
    }))

ws = websocket.WebSocketApp(
    'wss://ws.gox.exchange',
    on_message=on_message,
    on_open=on_open
)

# Usage
gold = get_gold_price()
print(f"Gold: ${gold['price_usd']}")

paxg = get_token_price('PAXG')
print(f"PAXG: ${paxg['price_usd']} ({paxg['change_pct_24h']}%)")

# Start WebSocket
ws.run_forever()
```

### React Hook

```typescript
import { useState, useEffect } from 'react';

function useGoldPrice() {
  const [price, setPrice] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // Initial fetch
    fetch('https://api.gox.exchange/v1/market/gold/spot')
      .then(res => res.json())
      .then(data => {
        setPrice(data);
        setLoading(false);
      });

    // WebSocket for real-time updates
    const ws = new WebSocket('wss://ws.gox.exchange');
    
    ws.onopen = () => {
      ws.send(JSON.stringify({
        action: 'subscribe',
        channel: 'gold_price'
      }));
    };

    ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.channel === 'gold_price') {
        setPrice(prev => ({ ...prev, ...data.data }));
      }
    };

    return () => ws.close();
  }, []);

  return { price, loading };
}

// Usage in component
function GoldPrice() {
  const { price, loading } = useGoldPrice();

  if (loading) return <div>Loading...</div>;

  return (
    <div>
      <h2>Gold Price</h2>
      <p>${price.price_usd}</p>
      <p>{price.change_pct_24h}% (24h)</p>
    </div>
  );
}
```

## Error Handling

### HTTP Status Codes

| Code | Meaning |
|------|---------|
| 200 | Success |
| 400 | Bad Request - Invalid parameters |
| 401 | Unauthorized - Invalid API key |
| 404 | Not Found - Resource doesn't exist |
| 429 | Too Many Requests - Rate limit exceeded |
| 500 | Internal Server Error |
| 503 | Service Unavailable - Maintenance |

### Error Response Format

```json
{
  "error": {
    "code": "INVALID_SYMBOL",
    "message": "Token symbol 'INVALID' not found",
    "details": {
      "supported_symbols": ["GOLD", "PAXG", "XAUT", "gXAU"]
    }
  },
  "timestamp": "2025-12-11T10:30:15Z"
}
```

## Rate Limiting

Headers in response:

```http
X-RateLimit-Limit: 100
X-RateLimit-Remaining: 95
X-RateLimit-Reset: 1702298460
```

When rate limited:

```json
{
  "error": {
    "code": "RATE_LIMIT_EXCEEDED",
    "message": "Rate limit exceeded. Try again in 45 seconds",
    "retry_after": 45
  }
}
```
